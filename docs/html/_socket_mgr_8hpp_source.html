50class SocketMgr

56 virtual ~SocketMgr()

66 virtual bool StartNetworkThreads(uint32_t threads = 1)
67 {
68 for (uint32_t i = 0; i < threads; i++) {
69 NetworkThreadPtr network_thr = std::make_shared<NetworkThread<SocketType>>();
70
71 if (network_thr == nullptr) {
72 HLog(error) << "Networking: Error in creating threads, SocketMgr::StartThreadForNetworks.";
73 return false;
74 }
75
76 network_thr->start();
77
78 _thread_map.insert(std::make_pair(i, network_thr));
79 }
80
81 return true;
82 }
83
87 virtual void stop_network()
88 {
89 /* Clear the thread map. */
90 for (auto it = _thread_map.begin(); it != _thread_map.end();) {
91 NetworkThreadPtr thr = it->second;
92 thr->finalize();
93 // Wait for thread to finalize all sockets.
94 while (thr->is_finalizing())
95 ;
96 thr->join();
97 HLog(info) << "Finalized network thread " << (void *) thr.get();
98 it = _thread_map.erase(it);
99 }
100 }
101
106 uint32_t GetNetworkThreadCount() const { return (uint32_t) _thread_map.size(); }
107
112 uint32_t SelectThreadWithMinConnections() const
113 {
114 uint32_t min_idx = 0;
115
116 for (auto &thr : _thread_map)
117 if (thr.second->connection_count() < NetworkThreadPtr(_thread_map.at(min_idx))->connection_count())
118 min_idx = thr.first;
119
120 return min_idx;
121 }

130 std::shared_ptr<SocketType> on_socket_open(std::shared_ptr<tcp::socket> const &socket, uint32_t thread_index)
131 {
132 std::shared_ptr<SocketType> new_socket = std::make_shared<SocketType>(++_last_socket_id, std::move(socket));
133
134 try {
135 // Add socket to thread.
136 NetworkThreadPtr(_thread_map.at(thread_index))->add_socket(new_socket);
137 } catch (boost::system::system_error const &error) {
138 HLog(error) << "Networking: Failed to retrieve client's remote address " << error.what();
139 }
140
141 return new_socket;
142 }
143
148 std::pair<std::shared_ptr<tcp::socket>, uint32_t> get_new_socket()
149 {
150 int min_idx = SelectThreadWithMinConnections();
151 return std::make_pair(NetworkThreadPtr(_thread_map.at(min_idx))->get_new_socket(), min_idx);
152 }
153
154private:
155 uint64_t _last_socket_id{0};
156 std::unordered_map<uint32_t, NetworkThreadPtr> _thread_map;
157};
