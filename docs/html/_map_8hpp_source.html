114 int d = r % (max - min + 1) + min;

131 MapCoords get_random_coordinates_in_walkable_area(uint16_t x, uint16_t y, int16_t xs, int16_t ys)
132 {
133 std::vector<MapCoords> available_cells;
134
135 assert(xs >= 0);
136 assert(ys >= 0);
137
138 for (int i = std::max(x - xs, 0); i < std::min(x + xs, (int) _width); i++) {
139 for (int j = std::max(y - ys, 0); j < std::min(y + ys,(int) _height); j++) {
140 if (i == x && j == y)
141 continue;
142 if (!has_obstruction_at(i, j))
143 available_cells.push_back(MapCoords(i, j));
144 }
145 }
146
147 if (available_cells.size() == 0)
148 return MapCoords(0, 0);
149
150 int rnd = rand() % available_cells.size();
151
152 return available_cells.at(rnd);
153 }

176template <class T>
177bool Horizon::Zone::Map::ensure_grid_for_entity(T *entity, MapCoords mcoords)
178{
179 std::string const &new_map_name = entity->map()->get_name();
180 GridCoords new_gcoords = mcoords.scale<MAX_CELLS_PER_GRID, MAX_GRIDS_PER_MAP>();
181
182 if (new_map_name.compare(get_name()) == 0 && entity->grid_coords() == new_gcoords)
183 return false;
184
185 if (entity->has_valid_grid_reference())
186 entity->remove_grid_reference();
187
188 entity->set_grid_coords(new_gcoords);
189
190 _gridholder.get_grid(new_gcoords.x(), new_gcoords.y()).add_object(entity);
191
192 return true;
193}

211template<class T, class CONTAINER>
212inline void Horizon::Zone::Map::visit(GridCoords const &lower_bound, GridCoords const &upper_bound, GridReferenceContainerVisitor<T, CONTAINER> &visitor)
213{
214 // Visit all grids in the map, from top to bottom, left to right
215 // This is done to ensure that the visitor is called in the same order as the grids are stored in the map
216 // This is important for the GridReferenceContainerVisitor, which will add the grid references to a container
217 // The container will then be iterated over in the same order as the grids were visited
218 // This ensures that the entities are visited in the container in the same order as they are stored in the map
219 // We visit all ranges - 0,0 to maximum grids in a map
220 // (the upper range as well, i.e. if the map is 10x10 grids, we visit 0,0 to 10,10 and not 9x9)
221 for (int y = upper_bound.y(); y >= lower_bound.y(); y--) {
222 for (int x = lower_bound.x(); x <= upper_bound.x(); x++) {
223 visit(x, y, visitor);
224 }
225 }
