161 void update()
162 {
163 _update_timer.expires_from_now(boost::posix_time::milliseconds(1));
164 _update_timer.async_wait(std::bind(&NetworkThread<SocketType>::update, this));
165
166 add_new_sockets();
167
168 // The following code removes all non-active from the active sockets list by shifting all
169 // active sockets to the left and then erasing the extra
170 _active_sockets.erase(std::remove_if(_active_sockets.begin(), _active_sockets.end(),
171 [this] (std::shared_ptr<SocketType> sock)
172 {
173 if (!sock->update() || is_finalizing()) {
174
175 if (sock->is_open())
176 sock->close_socket();
177
178 --_connections;
179
180 HLog(info) << "Socket closed in networking thread " << (void *) (_thread.get()) << " (Connections: " << _connections << ")";
181
182 return true;
183 }
184
185 return false;
186 }), _active_sockets.end());
187 }
188
196 void add_new_sockets()
197 {
198 if (is_finalizing())
199 return;
200
201 std::lock_guard<std::mutex> lock(_new_socket_queue_lock);
202
203 if (_new_socket_queue.empty())
204 return;
205
206 for (std::shared_ptr<SocketType> sock : _new_socket_queue) {
207 if (sock->is_open()) {
208 _active_sockets.push_back(sock);
209 // Start receiving from the socket.
210 sock->start();
211
212 ++_connections; // Increment network connections.
213
214 HLog(trace) << "A new socket has been added to network thread " << (void *) (_thread.get()) << " (Connections: " << _connections << ")";
215 }
216 }
217
218 _new_socket_queue.clear();
219 }
