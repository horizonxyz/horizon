77 void async_accept_with_callback(AcceptCallback callback)
78 {
79 std::shared_ptr<tcp::socket> socket;
80
81 uint32_t thread_index;
82
83 std::tie(socket, thread_index) = _socket_factory();
84
85 _acceptor.async_accept(*socket, [this, socket, callback, thread_index] (boost::system::error_code error)
86 {
87 if (!error) {
88 try {
89 socket->non_blocking(true);
90 callback(std::move(socket), thread_index);
91 } catch (boost::system::system_error const &err) {
92 HLog(error) << "Networking: AsyncAcceptor failed to initialize client's socket :" << err.what();
93 }
94 }
95
96 if (!_closed)
97 this->async_accept_with_callback(callback);
98 });
99 }

104 bool bind()
105 {
106 boost::system::error_code errorCode;
107
108 _acceptor.open(_endpoint.protocol(), errorCode);
109
110 if (errorCode) {
111 HLog(error) << "Failed to open acceptor " << errorCode.message().c_str();
112 return false;
113 }
114
115 _acceptor.bind(_endpoint, errorCode);
116
117 if (errorCode) {
118 HLog(error) << "Could not bind to " << _endpoint.address().to_string().c_str() << ":" << _endpoint.port() << " - "
119 << errorCode.message().c_str();
120 return false;
121 }
122
123 _acceptor.listen(boost::asio::socket_base::max_connections, errorCode);
124
125 if (errorCode) {
126 HLog(error) << "Failed to start listening on " << _endpoint.address().to_string().c_str() << ":" << _endpoint.port() << " " << errorCode.message().c_str();
127 return false;
128 }
129
130 return true;
131 }

136 void close()

162template <class T>
163void AsyncAcceptor::AsyncAccept()
164{
165 _acceptor.async_accept(_socket, [this] (boost::system::error_code error)
166 {
167 if (!error) {
168 try {
169 std::make_shared<T>(std::move(this->_socket))->start();
170 } catch (boost::system::system_error const &err) {
171 std::cerr << "Network Error: failed to retrieve client's remote address " << err.what() << std::endl;
172 }
173 }
174
175 if (!_closed)
176 this->AsyncAccept<T>();
177 });
178}
