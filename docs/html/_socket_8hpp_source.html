256 void read_handler_internal(boost::system::error_code error, size_t transferredBytes)
257 {
258 // If there is an error on the socket, we need to close it.
259 if (error) {
260 // If the socket is already closed, we don't need to do anything.
261 if (error.value() == boost::asio::error::eof) {
262 close_socket();
263 // If the connection was reset or timed out, we need to close the socket.
264 } else if (error.value() == boost::system::errc::connection_reset
265 || error.value() == boost::system::errc::timed_out) {
266 on_error();
267 close_socket();
268 // If there was any unknown error on the socket, we need to close it.
269 } else {
270 on_error();
271 close_socket();
272 }
273 // Retreive the error code and handle it accordingly.
274 switch (error.value())
275 {
276 case 2: // End of file
277 break;
278 default:
279 HLog(debug) << "Socket::read_handler_internal: " << error.value() << " (Code: " << error.message() << ").";
280 }
281 return;
282 }
283
284 if (transferredBytes > 0) {
285 // write_completed will move the write pointer forward, so we need to save the current position of the write pointer.
286 _read_buffer.write_completed(transferredBytes);
287 // read handler will process the data that has been written to the buffer.
288 read_handler();
289 }
290
291 // Invoke the next read operation.
292 async_read();
293 }

300 void write_handler_wrapper(boost::system::error_code /*error*/, std::size_t /*transferedBytes*/)
301 {
302 _is_writing_async = false;
303 handle_queue();
304 }
305
311 bool handle_queue()
312 {
313 boost::system::error_code error;
314
315 if (_write_queue.empty())
316 return false;
317
318 std::shared_ptr<ByteBuffer> to_send = _write_queue.front();
319
320 std::size_t bytes_sent = write_buffer_and_send(*to_send, error);
321
325 if (error == boost::asio::error::would_block || error == boost::asio::error::try_again)
326 return async_process_queue();
327
331 if (!error && bytes_sent < to_send->active_length()) {
332 to_send->read_completed(bytes_sent);
333 return async_process_queue();
334 }
335
336 // Pop the front element.
337 _write_queue.try_pop();
338
339 // Close if required.
340 if (_closing && _write_queue.empty())
341 close_socket();
342
343 // Return true if the queue is not empty. Indicating that the sending operation was successful.
344 return !_write_queue.empty() && bytes_sent;
345 }
