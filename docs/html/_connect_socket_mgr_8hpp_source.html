62 std::shared_ptr<Connector> start(std::string const &connection_name, Server *server, std::string const &connect_ip, uint16_t port, uint32_t connections = 1)
63 {
64 std::shared_ptr<Connector> connector;
65
66 if (!(connector = std::make_shared<Horizon::Networking::Connector>(connection_name, server, connect_ip, port))) {
67 HLog(error) << "ConnectSocketMgr::Start: " << connection_name << " failed to connect to tcp::" << connect_ip << "@" << port;
68 return nullptr;
69 }
70
71 if (!BaseSocketMgr::StartNetworkThreads(1)) {
72 HLog(error) << "ConnectSocketMgr::Start failed to start network threads.";
73 return nullptr;
74 }
75
76 // Set the socket factory. & Start attempting to connect.
77 connector->set_socket_factory(std::bind(&BaseSocketMgr::get_new_socket, this));
78 connector->connect_with_callback(
79 std::bind(&ConnectSocketMgr<SocketType>::on_socket_open,
80 this, std::placeholders::_1, std::placeholders::_2, std::placeholders::_3), connections);
81
82 return connector;
83 }

100 void on_socket_open(std::string const &conn_name, std::shared_ptr<tcp::socket> const &tcp_socket, uint32_t thread_index)
101 {
102 std::shared_ptr<SocketType> socket = SocketMgr<SocketType>::on_socket_open(std::move(tcp_socket), thread_index);

106 void add_socket_to_connections(std::string const &conn_name, std::shared_ptr<SocketType> sock)
107 {
108 std::unique_lock<std::shared_mutex> _connection_map_mtx;
109
110 _connection_map.emplace(conn_name, sock);
111 }
112
113 std::shared_ptr<SocketType> get_socket_from_connections(std::string const &conn_name)
114 {
115 std::shared_lock<std::shared_mutex> _connection_map_mtx;
116
117 try {
118 return _connection_map.at(conn_name);
119 } catch (std::out_of_range &e) {
120 return std::shared_ptr<SocketType>();
121 }
122 }
123
124 void remove_socket_from_connections(std::string const &conn_name)
125 {
126 std::unique_lock<std::shared_mutex> _connection_map_mtx;
127
128 auto socket = _connection_map.find(conn_name);
129
130 if (socket != _connection_map.end())
131 socket->second->delayed_close_socket();
132
133 _connection_map.erase(conn_name);
134 }
135
136
137private:
138 std::shared_mutex _connection_map_mtx;
139 ConnectionMap _connection_map;
140};
