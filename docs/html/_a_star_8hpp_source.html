47class Heuristic
48{
49 static MapCoords getDelta(MapCoords source_, MapCoords target_) { return MapCoords(abs(target_.x() - source_.x()), abs(target_.y() - source_.y())); }
50
51public:
52 static uint32_t manhattan(MapCoords source_, MapCoords target_)
53 {
54 auto delta = getDelta(source_, target_);
55 return static_cast<uint32_t>(10 * (delta.x() + delta.y()));
56 }
57
58 static uint32_t euclidean(MapCoords source_, MapCoords target_)
59 {
60 auto delta = getDelta(source_, target_);
61 return static_cast<uint32_t>(10 * std::sqrt(std::pow(delta.x(), 2) + std::pow(delta.y(), 2)));
62 }
63
64 static uint32_t octagonal(MapCoords source_, MapCoords target_)
65 {
66 auto delta = getDelta(source_, target_);
67 return 10 * (delta.x() + delta.y()) + (-6) * std::min(delta.x(), delta.y());
68 }
69};

136 CoordinateList findPath(MapCoords source_, MapCoords target_)
137 {
138 CoordinateList path;
139 std::shared_ptr<Node> current = nullptr;
140 NodeSet openSet, closedSet;
141 int searchStep = 0;
142
143 openSet.reserve(20);
144 closedSet.reserve(20);
145 openSet.push_back(std::make_shared<Node>(source_));
146
147 if (check_collision(target_.x(), target_.y()))
148 return path;
149
150 while (!openSet.empty() && searchStep < MAX_VIEW_RANGE) {
151 auto current_it = openSet.begin();
152
153 current = *current_it;
154
155 for (auto it = openSet.begin(); it != openSet.end(); it++) {
156 auto node = *it;
157 if (node->getScore() <= current->getScore()) {
158 current = node;
159 current_it = it;
160 }
161 }
162
163 if (current->coordinates == target_)
164 break;
165
166 closedSet.push_back(current);
167 openSet.erase(current_it);
168
169 for (uint32_t i = 0; i < directions; ++i) {
170
171 MapCoords newCoordinates(current->coordinates + direction[i]);
172
173 if (check_collision(newCoordinates.x(), newCoordinates.y())
174 || findNodeOnList(closedSet, newCoordinates)) {
175 continue;
176 }
177
178 newCoordinates.set_move_cost((i < 4) ? 10 : 14);
179 uint32_t totalCost = current->G + newCoordinates.move_cost();
180
181 std::shared_ptr<Node> successor = findNodeOnList(openSet, newCoordinates);
182 if (successor == nullptr) {
183 successor = std::make_shared<Node>(newCoordinates, current);
184 successor->G = totalCost;
185 successor->H = heuristic(successor->coordinates, target_);
186 openSet.push_back(successor);
187 } else if (totalCost < successor->G) {
188 successor->parent = current;
189 successor->G = totalCost;
190 }
191 }
192
193 searchStep++;
194 }
195
196 while (current != nullptr) {
197 path.push_back(current->coordinates);
198 current = current->parent;
199 }
200
201 releaseNodes(openSet);
202 releaseNodes(closedSet);
203
204 return path;
205 }
206
207private:
208 HeuristicFunction heuristic;
209 CollisionDetectionFunction check_collision;
210 CoordinateList direction = {
211 { 0, 1 }, { 1, 0 }, { 0, -1 }, { -1, 0 },
212 { -1, -1 }, { 1, 1 }, { -1, 1 }, { 1, -1 }
213 };
214 MapCoords worldSize;
215 uint32_t directions{0};
216};
