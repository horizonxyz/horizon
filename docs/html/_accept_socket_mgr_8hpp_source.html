65 virtual bool start(boost::asio::io_service &io_service, std::string const &listen_ip, uint16_t port, uint32_t threads = 1)
66 {
67 try {
68 _acceptor = std::make_unique<AsyncAcceptor>(io_service, listen_ip, port);
69 } catch (boost::system::system_error const &error) {
70 HLog(error) << "Exception caught in AcceptSocketMgr::Start (" << listen_ip.c_str() << ", " << port << ") " << error.what();
71 return false;
72 }
73
74 if (!BaseSocketMgr::StartNetworkThreads(threads)) {
75 HLog(error) << "AcceptSocketMgr failed to start network threads.";
76 return false;
77 }
78
79 _acceptor->set_socket_factory(std::bind(&BaseSocketMgr::get_new_socket, this));
80 _acceptor->async_accept_with_callback(std::bind(&AcceptSocketMgr<SocketType>::on_socket_open, this, std::placeholders::_1, std::placeholders::_2));
81
82 HLog(info) << "Networking initialized, listening on " << listen_ip << "@" << port << ".";
83 HLog(info) << "Maximum Network Threads: " << threads;
84
85 return true;
86 }

92 virtual void stop_network() override
93 {
94 if (_acceptor->is_open())
95 _acceptor->close();
96
97 BaseSocketMgr::stop_network();
98
99 _socket_map.clear();
100 }

109 void on_socket_open(std::shared_ptr<tcp::socket> const &socket, uint32_t thread_index)
110 {
111 std::shared_ptr<SocketType> new_socket = BaseSocketMgr::on_socket_open(std::move(socket), thread_index);
112
113 set_socket_for_management(new_socket);
114 }

121 void set_socket_for_removal(std::weak_ptr<SocketType> sock)
122 {
123 _socket_management_queue.push(std::make_pair(false, sock.lock()));
124 }
125
126 void set_socket_for_management(std::shared_ptr<SocketType> sock)
127 {
128 _socket_management_queue.push(std::make_pair(true, sock));
129 }
130
136 void update_socket_sessions(uint32_t time)
137 {
138 std::shared_ptr<std::pair<bool, std::shared_ptr<SocketType>>> sock_buf;
139
140 while ((sock_buf = _socket_management_queue.try_pop())) {
141 bool add = (*sock_buf).first;
142 std::shared_ptr<SocketType> socket = (*sock_buf).second;
143 auto socket_iter = _socket_map.find(socket->get_socket_id());
144
145 if (socket_iter != _socket_map.end()) {
146 if (!add)
147 _socket_map.erase(socket_iter);
148 } else if (add) {
149 _socket_map.emplace(socket->get_socket_id(), socket);
150 }
151 }
152
153 for (auto sock : _socket_map) {
154 if (sock.second->get_session() != nullptr) {
155 sock.second->update_session(time);
156 }
157 }
158 }
159
160private:
161 std::unique_ptr<AsyncAcceptor> _acceptor;
162 SocketMap _socket_map;
163 ThreadSafeQueue<std::pair<bool, std::shared_ptr<SocketType>>> _socket_management_queue;
164};
