409struct item_entry_data
410{
411 item_entry_data() { memset(this, 0, sizeof(*this)); }
412
413 bool is_equipment() { return type == IT_TYPE_WEAPON || type == IT_TYPE_ARMOR; }
414
415 bool is_stackable()
416 {
417 if (type == IT_TYPE_WEAPON || type == IT_TYPE_ARMOR
418 || type == IT_TYPE_PET_EGG || type == IT_TYPE_PET_ARMOR
419 || hire_expire_date > 0
420 )
421 return false;
422 return true;
423 }
424
425 bool operator == (item_entry_data const &right)
426 {
427 if (item_id != right.item_id
428 || bind_type != right.bind_type
429 || hire_expire_date != right.hire_expire_date
430 || unique_id != right.unique_id
431 || type != right.type
432 || info.is_favorite != right.info.is_favorite
433 )
434 return false;
435
436 return true;
437 }
438
439 bool operator != (item_entry_data const &right)
440 {
441 return !(*this == right);
442 }
443
444 // Assignment operator copy constructor
445 item_entry_data operator = (item_entry_data const &right)
446 {
447 if (storage_type == ITEM_STORE_INVENTORY)
448 index.inventory = right.index.inventory;
449 else if (storage_type == ITEM_STORE_STORAGE)
450 index.storage = right.index.storage;
451
452 item_id = right.item_id;
453 type = right.type;
454 amount = right.amount;
455 current_equip_location_mask = right.current_equip_location_mask;
456 actual_equip_location_mask = right.actual_equip_location_mask;
457 refine_level = right.refine_level;
458
459 for (int i = 0; i < MAX_ITEM_SLOTS; ++i)
460 slot_item_id[i] = right.slot_item_id[i];
461
462 hire_expire_date = right.hire_expire_date;
463 option_count = right.option_count;
464 ele_type = right.ele_type;
465
466 for (int i = 0; i < MAX_ITEM_OPTIONS; ++i)
467 {
468 option_data[i].index = right.option_data[i].index;
469 option_data[i].value = right.option_data[i].value;
470 option_data[i].param = right.option_data[i].param;
471 }
472 bind_type = right.bind_type;
473 info.is_favorite = right.info.is_favorite;
474 info.is_broken = right.info.is_broken;
475 info.is_identified = right.info.is_identified;
476 sprite_id = right.sprite_id;
477
478 unique_id = right.unique_id;
479 config = right.config;
480 return *this;
481 }
